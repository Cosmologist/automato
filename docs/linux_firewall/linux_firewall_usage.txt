### Отслеживаем входящие соединения с определенного интерфейса
#
# - Для этого смотрим все входящие пакеты
# - Понимаем что пакет относится к новому соединению (сигнатура пакета встречается первый раз)
# - Запоминаем сигнатуру
#
# Подробно:
# Добавляем правило в таблицу обработки меток сетевых пакетов - `--table mangle`
# 1. Отбор пакетов
#    - Находящиеся на этапе пред-маршрутизации - `--append PREROUTING`
#    - Пришедшие снаружи на интерфейс eno1 `--in-interface eno1`
# 2. Обработка пакетов
#    - Модуль отслеживания сетевых соединений connection track `--match conntrack`
#      (вводит понятие "соединение" внутри которого приходят и уходят пакеты -
#      вычисляя принадлежность пакета к соединению по хэшу параметров пакеты - Протокол-Source IP/Port, Destination IP/Port)
#    - Если пакет принадлежит новому соединению (хэш его параметров он видит впервые - это первый пакет) `--ctstate NEW`
#    - Запоминает ч что п`--jump CONNMARK` его указанной меткой `--set-mark 1` (уникальное 32-битное числовое значение)
iptables -t mangle -A PREROUTING -i eno1 -m conntrack --ctstate NEW -j CONNMARK --set-mark 1

### Отслеживаем пакеты в рамках соединения
#
# - Смотрим все исходящие пакеты `--append OUTPUT`
# - Если пакет имеет сигнатуру которая была зафиксирована ранее модулем `conntrack`
# - Ставим такую же метку и на этот пакет
iptables -t mangle -A OUTPUT -m conntrack --ctstate ESTABLISHED,RELATED -j CONNMARK --restore-mark

### Задаем правила маршрутизации пакетов с приоритетом
#
# - Пакеты помеченные определенной меткой `fwmark 1`
# - Должны идти по маршрутам указанным в таблице маршрутов 10002
# - Приоритет правила 32762
# - Сначала применятся правила с меньшим значением приоритета, потом с большим
ip rule add fwmark 1 table 10002 priority 32762

### Добавляем маршрут по-умолчанию в таблицу маршрутов
#
# - default - любой адрес назначения, который не попадает под другие маршруты
# - via - Через что отправляем - следующих хоп (шлюз)
# - dev - device - Интерфейс через который отправляем
# - table 10002 - в какую таблицу добавляем данный маршрут
ip route add default via 192.168.0.1 dev eno1 table 10002
